## FE có 2 nhiệm vụ chính trong API này
1. Tìm cách add map của VietMap vào web của mình cho user sài (tìm cách cho user bật GPS)
2. Khi BE trả ra Id, kinh độ vĩ độ của sân đó, FE tìm cách cách đánh marker cho sân đó
# API1 Bounding box -> Đánh marker
- Đầu tiên anh đọc phần request
	- Bên anh cung cấp tọa độ 4 điểm của màn hình user
- Rồi phần response
	- Tụi em trả ra  Id,  Kinh độ, Vĩ độ của từng Court bên trong 4 điểm mà anh gửi xuống
- Khi nhận lại Id, Kinh độ, Vĩ độ của từng Court đó, anh tiến hành đánh marker, đánh icon cho từng Court (cái này em ko rõ lắm, làm sao để tích cái bản đồ của VietMap và đánh marker, anh đọc bên docs của anh Tân nha)
- Khi user mà ấn marker đó, thì FE gửi courtId đó xuống cho BE bởi API em viết phần 2 á (`2. Khi customer ấn vào trang đó sẽ hiện ra các thông tin chi tiết của trang)`, BE sẽ gửi thông tin chi tiết sân đó lên cho FE

# Flow code

```CSharp
public async Task<Response.MapSearchResponse> SearchByBoundingBox(Request.BoundingBoxRequest request,  
    CancellationToken cancellationToken)  
{
//từ 4 tọa điể mà FE trả xuống
//BE dô DB query lấy tất cả các sân nằm trong khoảng đó để trả ra cho FE
    var markers = await _dbContext.Courts  
        .Where(x => x.Status == nameof(StatusCourt.Active)  
                && x.Latitude >= request.MinLat && x.Latitude <=request.MaxLat 
                && x.Longitude >= request.MinLon && x.Longitude <= request.MaxLon) 
        .Select(x => new Response.CourtMapItem  
        {  
            Id = x.Id,  
            Latitude = x.Latitude,  
            Longitude = x.Longitude,  
        }).ToListAsync(cancellationToken);  
    return new Response.MapSearchResponse  
    {  
        ListCourts = markers,  
        TotalCount = markers.Count,  
    };  
}
```

```CSharp
public async Task<Response.MapSearchResponse> SearchByRadius(Request.RadiusRequest request,  
    CancellationToken cancellationToken)  
{  
//Lấy tất cả các courts
    var allCourts = await _dbContext.Courts  
        .Where(x => x.Status == nameof(StatusCourt.Active))  
        .Select(x => new Response.CourtMapItem  
        {  
            Id = x.Id,  
            Latitude = x.Latitude,  
            Longitude = x.Longitude,  
        }).ToListAsync(cancellationToken);  
    if (allCourts.Count == 0)  
    {        return new Response.MapSearchResponse  
        {  
            ListCourts = allCourts,  
            TotalCount =  allCourts.Count,  
        };  
    } 
//gọi API của VietMap tính khoảng cách từ user đến từng court 
//ta được một list các khoảng cách
    var distances = await GetDistanceFromMatrix(request.Latitude,                                                          request.Longitude, allCourts);  
//Filter theo bán kính
    var markers = new  List<Response.CourtMapItem>();  
    for (int i = 0; i < allCourts.Count; i++)  
    {        if (distances[i] / 1000.0 <= request.RadiusKm)  
        {            markers.Add(allCourts[i]);  
        }    }  
    return new Response.MapSearchResponse  
    {  
        ListCourts = markers,  
        TotalCount = markers.Count,  
    };  
}
//Gọi API của VietMap để tính khoảng cách
//đầu vào là tọa độ của user, với danh sách các courts
//đầu ra là list các khoảng cách của User đến từng court đó
    private async Task<List<double>> GetDistanceFromMatrix(  
    decimal userLat, decimal userLon, List<Response.CourtMapItem> courts)  
{ 
//tạo một thằng client để nó gọi API VietMap
    var client = _httpClientFactory.CreateClient("VietMap");
//URL gọi API VietMap  
    var url = $"{_mapOptions.BaseUrl}/matrix?api-version=1.1&apikey=                                {_mapOptions.ApiKey}"  + $"&point={userLat},{userLon}";điểm first  
//đối với từng court thì thêm điểm của court đó vào    
    foreach (var court in courts)  
        url += $"&point={court.Latitude},{court.Longitude}";  
//user là nguồn (0), từ user tính đến các điểm của các courts
    url += "&sources=0";
//yêu cầu VietMap trả về khoảng cách (met)
    url += "&annotation=distance";
//lấy index của tất cả các court  
    url += $"&destinations={string.Join(";", Enumerable.Range(1, courts.Count))}"; 
    
//Gửi HTTP Get đến VietMap bằng url bởi thằng Client
    var response = await client.GetAsync(url);
//nếu gọi API vietmap bị lỗi -> trả về danh sách khoảng cách cực lớn để loại bỏ
//nếu ko có dòng bắt lỗi này thì nếu gọi API bị lỗi sẽ dẫn đến crash ở parse JSON
    if (!response.IsSuccessStatusCode)  
        return Enumerable.Repeat(double.MaxValue, courts.Count).ToList(); 
//lấy nội dung trả về dụng string
    var json = await response.Content.ReadAsStringAsync();  
//convert string sang json
    using var doc = JsonDocument.Parse(json);
//lúc này trong phần body sẽ có khoảng cách của source(0) là từ user 
    var distancesArray = doc.RootElement.GetProperty("distance")[0];  
//và add nó vô    
    var result = new List<double>();  
    foreach (var item in distancesArray.EnumerateArray())  
        result.Add(item.GetDouble());  
    return result;  
}
```

```CSharp
public async Task<Response.MapSearchResponse> SearchByText(Request.SearchByTextRequest request,  
    CancellationToken cancellationToken)  
{
//chuyển text thành kinh độ và vĩ độ  
    var coordinate = await GeocodeText(request.Text);
//nếu null thì return luôn  
    if (coordinate == null)  
    {        return new Response.MapSearchResponse  
        {  
            ListCourts = new(),  
            TotalCount = 0  
        };  
    } 
//nếu ok thì trả ra kinh độ và vĩ độ, rồi gọi làm hàm searchByRadius ở trên làm lại 
    return await SearchByRadius(new Request.RadiusRequest  
    {  
        Latitude = coordinate.Value.Latitude,  
        Longitude = coordinate.Value.Longitude,  
        RadiusKm = request.RadiusKm  
    }, cancellationToken);  
}

//Hàm này có nghiệm vụ giúp chuyển từ địa chỉ dạng text -> trả về kinh độ vĩ độ
private async Task<(decimal Latitude, decimal Longitude)?> GeocodeText(  
    string text)  
{ 
//tạo client
    var client = _httpClientFactory.CreateClient("VietMap");  
//url api của VietMap
    var url = $"{_mapOptions.BaseUrl}/search?api-version=1.1" +  
              $"&apikey={_mapOptions.ApiKey}" +  
              $"&text={Uri.EscapeDataString(text)}";  
//Uri.EscapeDataString(text) mã hóa text để tránh lỗi url
//gửi request
    var response = await client.GetAsync(url);  
//nếu lỗi thì trả về null
    if (!response.IsSuccessStatusCode) return null;  
//đọc nội dug trả về
    var json = await response.Content.ReadAsStringAsync(); 
//chuyển thành json object
    using var doc = JsonDocument.Parse(json);
//check thử api trả ra có phải array không, và có phần tử không 
    if (doc.RootElement.ValueKind == JsonValueKind.Array &&  //tránh lỗi format
        doc.RootElement.GetArrayLength() > 0)  //tránh craassh khi lấy phần tử 0
    {
//lúc này api trả ra nhiều kết quả, phần tử đầu liên có kết quả chính xác nhất     
    var first = doc.RootElement[0];  
    if(first.TryGetProperty("lat", out var lat) &&  
           first.TryGetProperty("lon", out var lon))  
    return ((decimal)lat.GetDouble(), (decimal)lon.GetDouble());  
    }    
    return null;  
}
```
